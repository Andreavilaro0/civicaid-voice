**Step 3: Crear `src/i18n/provider.tsx`**

Wrapper client que carga mensajes dinamicamente segun `language` prop:

```typescript
"use client";

import { NextIntlClientProvider } from "next-intl";
import { useEffect, useState } from "react";
import type { Language } from "@/lib/types";

import esMessages from "@/messages/es.json";
import frMessages from "@/messages/fr.json";

const allMessages: Record<Language, typeof esMessages> = {
  es: esMessages,
  fr: frMessages,
};

interface I18nProviderProps {
  language: Language;
  children: React.ReactNode;
}

export default function I18nProvider({ language, children }: I18nProviderProps) {
  const [messages, setMessages] = useState(allMessages[language]);

  useEffect(() => {
    setMessages(allMessages[language]);
  }, [language]);

  return (
    <NextIntlClientProvider locale={language} messages={messages}>
      {children}
    </NextIntlClientProvider>
  );
}
```

const messages = { es: esMessages, fr: frMessages };

function getErrorMessage(category: string, language: Language) {
  const m = messages[language].Errors;
  const msgKey = `${category}_message` as keyof typeof m;
  const actKey = `${category}_action` as keyof typeof m;
  return {

  };
}
